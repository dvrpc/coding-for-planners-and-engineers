#filtering the tracts
finalTracts <-
  filter(tracts, as.numeric(substr(
    tracts$GEOID, start = 1, stop = 5
  )) %in% dvrpc_counties)
finalTracts$GEOID <- as.numeric(finalTracts$GEOID)


#exporting your files
